# 什么是awk？
AWK是一种解释性编程语言，主要用于文本处理。之所以叫AWK是因为其取了三位创始人Alfred Aho，Peter Weinberger,和Brian Kernighan的Family Name的首字符。按行读取数据，按行处理数据。


# awk可以做什么？
读取文件做一定的程序化处理

# 如何安装awk？
在MacOS、Ubuntu等系统中自带；
在windows系统下，建议使用scoop，winget等包管理器进行安装，或者官网下载安装。

# 常见语法
awk的语法分为4个部分：awk 变量设置(非必要)  主体脚本(必要)  文件(非必要)
变量设置：我们可以设置切分格式、变量等
主体脚本：'BEGIN{各种处理语句}END'
文件：要处理的文件信息
## 输入输出
```shell
# 设定变量name并在后面输出
awk -v name=TomAndJerry 'BEGIN{printf name}'

awk -v name=TomAndJerry '{printf name}'
```

## 行操作
```shell
# 打印文件每行分割后的第一个
# $0 代表整个行 $1代表分割的第一个，以此内推
awk '{print $1}' a.txt

# 打印包含a字符的行的所有信息
awk '/a/ {print $0}' a.txt

# 打印包含a字符的行的行数
awk '/a/{++cnt} END {print "Count = ", cnt}' a.txt

# 以,为分隔符 分割每一行数据(awk默认是以空格分割)
echo "啊,b,c" | awk -F "," '{print $2}'

echo "a,b,c" | awk -v FS="," '{print $2}'

# NF 表示分割后每一行的元素个数
# 输出分割后元素大于2个的行
echo "a b c" | awk 'NF>2'

# NR 行号，当前是第一行
# FNR 当前文件中的第几行，多用于多文件的处理
awk 'print NR' a.txt

# OFS 与ORS  
# OFS是输出分割符号，默认是空格
# ORS是输出行分割符号，默认为换行符

# 字符串的拼接
awk 'BEGIN { str1 = "Hello, "; str2 = "World"; str3 = str1 str2; print str3 }'

# 数组
awk 'BEGIN { arr[0] = 1; arr[1] = 2; arr[2] = 3; for (i in arr) printf "arr[%d] = %d\n", i, arr[i] }'

```

## if判断
```shell
# 判断两数是否相等
awk 'BEGIN { a = 10; b = 10; if (a == b) print "a == b" }'

# 复杂逻辑处理
awk 'BEGIN { num = 5; if (num >= 0 && num <= 7) printf "%d is in octal format\n", num }'

# 三元两次方程
awk 'BEGIN { a = 10; b = 20; (a > b) ? max = a : max = b; print "Max =", max }'

```

echo -e "One Two\nOne Two Three\nOne Two Three Four" | awk 'NF > 2'
echo "(8). NR: 行号，如执行："
echo -e "One Two\nOne Two Three\nOne Two Three Four" | awk 'NR < 3'
echo "(9). FNR: 与NR相似，不过在处理多文件时更有用，获取的行号相对于当前文件."
echo "(10). OFMT: 输出格式数字,默认值为%.6g，执行："
awk 'BEGIN {print "OFMT = " OFMT}'
echo "(11). OFS: 输出字段分隔符，默认为空格,执行:"
awk 'BEGIN {print "OFS = " OFS}' | cat -vte
echo "(12). ORS: 输出行分隔符,默认值为换行符,执行:"
awk 'BEGIN {print "ORS = " ORS}' | cat -vte
echo "(13). RLENGTH: 代表了match函数匹配的字符串长度,如执行："
awk 'BEGIN { if (match("One Two Three", "re")) { print RLENGTH } }'
echo "(14). RS: 输入记录分隔符,默认值为换行符,如执行："
awk 'BEGIN {print "RS = " RS}' | cat -vte
echo "(15). RSTART: match函数匹配的第一次出现位置,如执行:"
awk 'BEGIN { if (match("One Two Three", "Thre")) { print RSTART } }'
echo "(16). SUBSEP: 数组子脚本的分隔符，默认为^\，如执行:"
awk 'BEGIN { print "SUBSEP = " SUBSEP }' | cat -vte
echo "(17). $0: 代表了当前行,如执行:"
awk '{print $0}' marks.txt
echo "(18). $n: 当前行中的第n个字段,字段间由FS分隔,如执行："
awk '{print $3 "\t" $4}' marks.txt
 
echo; echo "GNU AWK的变量："

echo "(2). BINMODE: 在非POSIX系统上指定对所有的文件I/O采用二进制模式"

echo "(7). PROCINFO: 包含进程信息的关联数组，例如UID，进程ID等,如执行:"
